Image Slider

Code:

<!-- Image Slider Structure -->
<div class="slider-container w-full h-64 bg-gray-200">
    <div class="slider-wrapper h-full">
        <img src="https://placehold.co/800x400/FFCCBC/374151?text=Slide+1" alt="Slide 1">
        <img src="https://placehold.co/800x400/D1C4E9/374151?text=Slide+2" alt="Slide 2">
        <img src="https://placehold.co/800x400/C8E6C9/374151?text=Slide+3" alt="Slide 3">
    </div>
    <button class="slider-button left" onclick="moveSlide(-1)">&#10094;</button>
    <button class="slider-button right" onclick="moveSlide(1)">&#10095;</button>
</div>

<!-- Corresponding JavaScript (place at end of body) -->
<script>
    let currentSlide = 0;
    const slides = document.querySelectorAll('.slider-wrapper img');
    const totalSlides = slides.length;

    function showSlide(index) {
        if (index >= totalSlides) {
            currentSlide = 0;
        } else if (index < 0) {
            currentSlide = totalSlides - 1;
        } else {
            currentSlide = index;
        }
        const sliderWrapper = document.querySelector('.slider-wrapper');
        sliderWrapper.style.transform = `translateX(${-currentSlide * 100}%)`;
    }

    function moveSlide(direction) {
        showSlide(currentSlide + direction);
    }

    document.addEventListener('DOMContentLoaded', () => {
        showSlide(0);
    });
</script>

<!-- Accordion Menu Structure -->
<div class="space-y-4">
    <div class="accordion-item">
        <div class="accordion-header" onclick="toggleAccordion(this)">
            <span>Section Title 1</span>
            <span>&#9660;</span>
        </div>
        <div class="accordion-content">
            <p>Content for section 1 goes here.</p>
        </div>
    </div>
    <div class="accordion-item">
        <div class="accordion-header" onclick="toggleAccordion(this)">
            <span>Section Title 2</span>
            <span>&#9660;</span>
        </div>
        <div class="accordion-content">
            <p>Content for section 2 goes here.</p>
        </div>
    </div>
</div>

<!-- Corresponding JavaScript (place at end of body) -->
<script>
    function toggleAccordion(header) {
        const item = header.closest('.accordion-item');
        if (item) {
            document.querySelectorAll('.accordion-item.active').forEach(activeItem => {
                if (activeItem !== item) {
                    activeItem.classList.remove('active');
                }
            });
            item.classList.toggle('active');
        }
    }
</script>
